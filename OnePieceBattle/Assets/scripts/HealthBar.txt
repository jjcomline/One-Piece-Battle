using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class HealthBar : MonoBehaviour
{
    public Transform bar;

    public Animator animator;
    
    private void Start() {
        //bar_color = GetComponent<Renderer>();
    }

    public void SetSize(int sizeNormalized) {         
        if (sizeNormalized < 71 && sizeNormalized > 40) {
            animator.SetInteger("Danger_1", sizeNormalized);
        }
        if (sizeNormalized <= 40 && sizeNormalized > 25 ){
            animator.SetInteger("Danger_1", sizeNormalized);
        }
        if (sizeNormalized <= 25){
            animator.SetInteger("Danger_1", sizeNormalized);
        }
        bar.localScale = new Vector3 ((float)sizeNormalized / 100f, 1f);
    }

    public void resetHealth() {
        bar.localScale = new Vector3(1f, 1f);
    } 
}
